\section{\acs{API} Code-Beispiele}
\begin{lstlisting}[language=Clojure,caption=\textit{read-chan} Hilfsfunktion,label=lst:readchan]
(defn read-chan
  ([c] (read-chan "" c))
  ([str c]
  (thread
    (loop [val (<!! c)]
      (when (not= val nil)
        (println str val)
        (recur (<!! c))))
      (println "END."))))
\end{lstlisting}

\subsection{Timeout}
\begin{lstlisting}[language=Clojure,caption=Timeout,label=lst:timeout]
(let [c (timeout 4000)]
  (put! c "test")
  (thread
  (loop [val (<!! c)]
    (when (not= val nil)
      (println val)
      (recur (<!! c))))
    (println "timeout")))
\end{lstlisting}
\subsection{Filter}
\begin{lstlisting}[language=Clojure,caption=Filter,label=lst:filter]
(let [out (chan)
      c (filter> string? out)]
  (thread
    (>!! c "abc")
    (>!! c 123))
  (read-chan out))
\end{lstlisting}
